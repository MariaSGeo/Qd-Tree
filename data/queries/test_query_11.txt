select
	100.00 * sum(case
		when p_type like 'PROMO%'
			then l_extendedprice * (1 - l_discount)
		else 0
	end) / sum(l_extendedprice * (1 - l_discount)) as promo_revenue
from
	lineitem
where
	l_shipdate >= date '1996-07-01'
	and l_shipdate < date '1996-07-01' + interval '1' month;
where rownum <= -1;
